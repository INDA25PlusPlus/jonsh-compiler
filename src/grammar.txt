<program> ::= <statement>+
<terminator> ::= ";"
<spacing> ::= ( " " | "\n" | "\t" )*
<print> ::= "return(" <exp> ")"

<number> ::= ( [1-9] [0-9]* ) | "0"
<variable> ::= [a-z] ( [a-z] | [0-9] )*
<exp> ::= <variable> | <number> 

<definition> ::= <variable> ( "=" | " = " ) ( <exp> | <operation> )
<mathop> ::=  "+" | "-" | "*" | "/"
<compare> ::= ( <exp> <compareop> <exp> ) | ( <exp> " " <compareop> " " <exp> )
<compareop> ::= "<" | ">" | "<=" | ">=" | "==" | "!=" 
<operation> ::= <exp> ( ( <mathop> | " " <mathop> " " )  <exp> )+ 

<statementif> ::= "if(" <compare> ")" <spacing> <block> (<spacing> <statementelif>)* ( <spacing> <statementelse> )?
<statementelif> ::= "elif(" <compare> ")" <spacing> <block>
<statementelse> ::= "else" <spacing> <block>

<statementwhile> ::= "while(" <compare> ")" <spacing> <block>
<statementfor> ::= "for(" <variable> "," <number> ")" <spacing> <block>

<block> ::= "{" <statement>+ "}"
<statement> ::=  <spacing> ( ( <print> | <definition> | <statementif> | <statementwhile> | <statementfor> ) <terminator> ) <spacing>
